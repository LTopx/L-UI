### 尺寸大小

<Code component={<Size />}>
<Code.Description>
  提供三种尺寸的大小：`sm` , `base` , `lg`
</Code.Description>
```tsx filename="Accordion.tsx" copy
import {
  Accordion,
  Tabs,
  type AccordionItem,
  type TabsOption,
} from "@ltopx/lx-ui";

export default function () {
  const items: AccordionItem[] = [
    {
      label: "111",
      value: "111",
      children: <div>111</div>,
    },
    {
      label: "222",
      value: "222",
      children: <div>222</div>,
    },
    {
      label: "333",
      value: "333",
      children: <div>333</div>,
    },
  ];
  
  const tabOptions: TabsOption[] = [
    {
      label: "sm",
      value: "sm",
      children: <Accordion items={items} size="sm" />,
    },
    {
      label: "base",
      value: "base",
      children: <Accordion items={items} />,
    },
    {
      label: "lg",
      value: "lg",
      children: <Accordion items={items} size="lg" />,
    },
  ];
  
  return <Tabs options={tabOptions} defaultActiveTab="sm" />;
}
```
</Code>

## API

### Accordion Props

| 属性             | 描述         | 类型                   | 默认值 |
| ---------------- | ------------ | ---------------------- | ------ |
| className        | 节点类名     | `string`               | -      |
| itemClassName    | 选项节点类名 | `string`               | -      |
| triggerClassName | 选项标题类名 | `string`               | -      |
| contentClassName | 选项内容类名 | `string`               | -      |
| size             | 尺寸大小     | `sm` \| `base` \| `lg` | `base` |
| items            | 选项内容列表 | `AccordionProps[]`     | -      |

### AccordionProps Props

| 属性     | 描述         | 类型              | 默认值 |
| -------- | ------------ | ----------------- | ------ |
| label    | 选项标题     | `React.ReactNode` | -      |
| value    | 选项唯一键   | `string`          | -      |
| children | 选项显示内容 | `React.ReactNode` | -      |
