### 基础用法

<Code 
  description='通过键盘或鼠标输入内容。' 
  component={<Basic />}
>
```tsx filename="Input.tsx" copy
import { Input } from "@ltopx/lx-ui";

export default function () {
  const [value, setValue] = React.useState<string | number>("");
  
  return (
    <div className="flex flex-wrap gap-2">
      <Input
        className="w-52"
        placeholder="请输入"
        allowClear
        value={value}
        onChange={setValue}
      />
      <Input className="w-52" placeholder="请输入" allowClear type="password" />
    </div>
  );
}
```
</Code>

### 手动触发事件

<Code 
  description='继承原生input的所有触发事件' 
  component={<Trigger />}
>
```tsx filename="Input.tsx" copy
import { Input, Button } from "@ltopx/lx-ui";

export default function () {
  const inputRef = React.useRef<HTMLInputElement>(null);
  
  const onFocus = () => inputRef.current?.focus();
  
  return (
    <div className="flex flex-wrap gap-2">
      <Button type="primary" onClick={onFocus}>
        聚焦Input
      </Button>
      <Input className="w-52" placeholder="请输入" allowClear ref={inputRef} />
    </div>
  );
}
```
</Code>

## API

### Input Props

| 属性        | 描述                                                                                                                                        | 类型                   | 默认值 |
| ----------- | ------------------------------------------------------------------------------------------------------------------------------------------- | ---------------------- | ------ |
| className   | 结构类名                                                                                                                                    | `string`               | -      |
| type        | input 类型，与原生 input 标签的 type 属性相同。见 [MDN](https://developer.mozilla.org/zh-CN/docs/Web/HTML/Element/input#%E5%B1%9E%E6%80%A7) | `string`               | `text` |
| size        | 尺寸大小                                                                                                                                    | `sm` \| `base` \| `lg` | `base` |
| allowClear  | 是否允许清空                                                                                                                                | boolean                | -      |
| disabled    | 是否禁用                                                                                                                                    | boolean                | -      |
| min         | 最小值                                                                                                                                      | number                 | -      |
| max         | 最大值                                                                                                                                      | number                 | -      |
| maxLength   | 输入框最大输入的长度                                                                                                                        | number                 | -      |
| step        | 数字变化步长                                                                                                                                | number                 | 1      |
| placeholder | 默认占位文本                                                                                                                                | `string`               | -      |
| value       | 输入框的值，受控模式                                                                                                                        | `string` \| `number`   | -      |
| onFocus     | 获得焦点时的回调                                                                                                                            | `() => void`           | -      |
| onBlur      | 失去焦点时的回调                                                                                                                            | `() => void`           | -      |
| onChange    | 输入时的回调                                                                                                                                | `(value: any) => void` | -      |
| onEnter     | 按下回车键的回调                                                                                                                            | `(value: any) => void` | -      |

### Input Methods

| 属性  | 描述     | 类型                               | 默认值 |
| ----- | -------- | ---------------------------------- | ------ |
| focus | 获取焦点 | `(options?: FocusOptions) => void` | -      |
| blur  | 取消焦点 | `() => void`                       | -      |
