### 基础用法

<Code 
  description='基础用法' 
  component={<Basic />}
>
```tsx filename="Select.tsx" copy
import React from "react";
import { Select, type SelectOption } from "@ltopx/lx-ui";

export default function () {
  const options: SelectOption[] = [
    { label: "1", value: "1" },
    { label: "2", value: "2", disabled: true },
    { label: "3", value: "3" },
  ];
  
  const [value, setValue] = React.useState<number>(1);
  
  const onChange = (value: string) => {
    console.log(value, "value");
  };
  
  const onChangeValue = (value: number) => {
    setValue(value);
  };
  
  return (
    <div className="flex flex-wrap gap-3">
      <Select
        className="w-60"
        placeholder="请选择"
        options={options}
        onChange={onChange}
      />
      <Select
        className="w-60"
        placeholder="请选择"
        options={options}
        disabled
      />
      <Select
        className="w-60"
        placeholder="请选择"
        loading
        options={[1, 2, 3]}
        value={value}
        onChange={onChangeValue}
      />
    </div>
  );
}
```
</Code>

### 自定义渲染

<Code component={<RenderLabel />}>
<Code.Description>
  使用 `renderLabel`， 可以自定义渲染节点内容
</Code.Description>
```tsx filename="Select.tsx" copy {20}
import { Select, type SelectOption } from "@ltopx/lx-ui";

export default function () {
  const options: SelectOption[] = [
    { label: "1", value: "1" },
    { label: "2", value: "2", disabled: true },
    { label: "3", value: "3" },
  ];
  
  const renderLabel = (item: SelectOption) => {
    return <div>😄{item.label}😄</div>;
  };
  
  return (
    <div className="flex gap-3">
      <Select
        className="w-60"
        placeholder="请选择"
        options={options}
        renderLabel={renderLabel}
      />
    </div>
  );
}
```
</Code>

### 尺寸大小

<Code component={<Size />}>
<Code.Description>
  提供三种尺寸的大小：`sm` , `base` , `lg`
</Code.Description>
```tsx filename="Select.tsx" copy {27,52}
import {
  Select,
  Tabs,
  type SelectOption,
  type TabsOption,
} from "@ltopx/lx-ui";

export default function () {
  const options: SelectOption[] = [
    { label: "1", value: "1" },
    { label: "2", value: "2", disabled: true },
    { label: "3", value: "3" },
  ];
  
  const onChange = (value: string) => {
    console.log(value, "value");
  };
  
  const tabOptions: TabsOption[] = [
    {
      label: "sm",
      value: "sm",
      children: (
        <Select
          className="w-60"
          placeholder="请选择"
          size="sm"
          options={options}
          onChange={onChange}
        />
      ),
    },
    {
      label: "base",
      value: "base",
      children: (
        <Select
          className="w-60"
          placeholder="请选择"
          options={options}
          onChange={onChange}
        />
      ),
    },
    {
      label: "lg",
      value: "lg",
      children: (
        <Select
          className="w-60"
          placeholder="请选择"
          size="lg"
          options={options}
          onChange={onChange}
        />
      ),
    },
  ];
  
  return <Tabs options={tabOptions} defaultActiveTab="sm" />;
}
```
</Code>

## API

### Select Props

| 属性         | 描述                             | 类型                                | 默认值 |
| ------------ | -------------------------------- | ----------------------------------- | ------ |
| className    | 节点类名                         | `string`                            | -      |
| placeholder  | 默认占位文本                     | `string`                            | -      |
| disabled     | 是否禁用                         | `boolean`                           | -      |
| loading      | 是否正在加载下拉列表数据         | `boolean`                           | -      |
| size         | 尺寸大小                         | `sm` \| `base` \| `lg`              | `base` |
| options      | 下拉选项列表                     | `SelectOption[]`                    | -      |
| renderLabel  | 自定义渲染列表显示内容           | `(value: any) => React.ReactNode`   | -      |
| defaultValue | 默认选中条目                     | `string` \| `number`                | -      |
| value        | 受控模式下选中的条目             | `string` \| `number`                | -      |
| onChange     | 选中条目发生变化时，会触发此方法 | `(value: string \| number) => void` | -      |

### SelectOption Props

| 属性     | 描述           | 类型                 | 默认值 |
| -------- | -------------- | -------------------- | ------ |
| label    | 选项展示内容   | `React.ReactNode`    | -      |
| value    | 选项选中时的值 | `string` \| `number` | -      |
| disabled | 是否禁用       | `boolean`            | false  |
