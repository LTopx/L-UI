### Loading

<Code component={<Loading />}>
<Code.Description>
  You can set the `loading` to make the button display a loading state.
</Code.Description>
```tsx filename="Button.tsx" copy
import React from "react";
import { Button } from "@ltopx/lx-ui";

export default function () {
  const [loading, setLoading] = React.useState(false);
  
  const onClick1 = () => {
    setLoading(true);
    setTimeout(() => {
      setLoading(false);
    }, 2000);
  };
  
  return (
    <div className="flex flex-col gap-4">
      <div className="flex flex-wrap gap-x-3 gap-y-2">
        <Button type="primary" loading>
          Loading
        </Button>
        <Button type="success" loading>
          Loading
        </Button>
        <Button type="primary" rounded loading>
          Loading
        </Button>
      </div>
      <div className="flex">
        <Button type="primary" loading={loading} onClick={onClick1}>
          {loading ? "loading" : "Primary"}
        </Button>
      </div>
    </div>
  );
}
```

</Code>

## API

### Button Props

| Name      | Description                                                                         | Type                                                                   | Default   |
| --------- | ----------------------------------------------------------------------------------- | ---------------------------------------------------------------------- | --------- |
| className | Additional css class                                                                | `string`                                                               | -         |
| type      | Button types                                                                        | `default` \| `primary` \| `success` \| `warning` \| `danger` \| `link` | `default` |
| disabled  | Whether is disabled                                                                 | `boolean`                                                              | -         |
| size      | Button size                                                                         | `sm` \| `base` \| `lg`                                                 | `base`    |
| loading   | Whether is in loading                                                               | `boolean`                                                              | -         |
| block     | Whether it occupies the entire width of the parent container.                       | `boolean`                                                              | -         |
| outline   | Whether to display the button in a bordered form                                    | `boolean`                                                              | -         |
| rounded   | Whether to display the buttons in a fully rounded shape.                            | `boolean`                                                              | -         |
| icon      | Set button icon, located on the left side of the text (if there is any).            | `React.ReactNode`                                                      | -         |
| href      | Add a hyperlink, by setting this attribute, the button will behave like an `a` tag. | `string`                                                               | -         |
| target    | As the target attribute of the `a` tag. Only effective when href exists.            | `HTMLAttributeAnchorTarget`                                            | -         |
| onClick   | Click button callback.                                                              | `(e: Event) => void`                                                   | -         |
