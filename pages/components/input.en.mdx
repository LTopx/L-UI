### Basic usage

<Code 
  description='Enter content through the keyboard or mouse.' 
  component={<Basic />}
>
```tsx filename="Input.tsx" copy
import React from 'react';
import { Input } from "@ltopx/lx-ui";

export default function () {
  const [value, setValue] = React.useState<string | number>("123");
  
  return (
    <div className="flex flex-wrap gap-2">
      <Input
        className="w-52"
        placeholder="Please input"
        allowClear
        value={value}
        onChange={setValue}
      />
      <Input
        className="w-52"
        placeholder="Please input"
        allowClear
        defaultValue="hello world"
      />
      <Input
        className="w-52"
        placeholder="Please input"
        allowClear
        type="password"
      />
    </div>
  );
}
```
</Code>

### Manually trigger event

<Code 
  description='Inherit all triggering events from the native input.' 
  component={<Trigger />}
>
```tsx filename="Input.tsx" copy
import React from "react";
import { Input, Button } from "@ltopx/lx-ui";

export default function () {
  const inputRef = React.useRef<HTMLInputElement>(null);
  
  const onFocus = () => inputRef.current?.focus();
  
  return (
    <div className="flex flex-wrap gap-2">
      <Button type="primary" onClick={onFocus}>
        Focus input
      </Button>
      <Input className="w-52" placeholder="Please input" allowClear ref={inputRef} />
    </div>
  );
}
```
</Code>

## API

### Input Props

| Name         | Description                                                                                                                                                              | Type                   | Default |
| ------------ | ------------------------------------------------------------------------------------------------------------------------------------------------------------------------ | ---------------------- | ------- |
| className    | Additional css class                                                                                                                                                     | `string`               | -       |
| type         | The input type, is the same as the type attribute of the native input tag. See [MDN](https://developer.mozilla.org/zh-CN/docs/Web/HTML/Element/input#%E5%B1%9E%E6%80%A7) | `string`               | `text`  |
| size         | Input size                                                                                                                                                               | `sm` \| `base` \| `lg` | `base`  |
| allowClear   | Whether allowed to clear                                                                                                                                                 | boolean                | -       |
| disabled     | Whether is disabled                                                                                                                                                      | boolean                | -       |
| min          | Minimum value                                                                                                                                                            | number                 | -       |
| max          | Maximum value                                                                                                                                                            | number                 | -       |
| maxLength    | The maximum input length of the text.                                                                                                                                    | number                 | -       |
| step         | The number to which the current value is increased or decreased. It can be an integer or decimal                                                                         | number                 | 1       |
| placeholder  | Placeholder of element                                                                                                                                                   | `string`               | -       |
| defaultValue | Default value                                                                                                                                                            | `string` \| `number`   | -       |
| value        | The current value                                                                                                                                                        | `string` \| `number`   | -       |
| onFocus      | Callback when focus.                                                                                                                                                     | `() => void`           | -       |
| onBlur       | Callback when losing focus                                                                                                                                               | `() => void`           | -       |
| onChange     | Callback when entering                                                                                                                                                   | `(value: any) => void` | -       |
| onEnter      | Press Enter key callback.                                                                                                                                                | `(value: any) => void` | -       |

### Input Methods

| Name  | Description | Type                               | Default |
| ----- | ----------- | ---------------------------------- | ------- |
| focus | Get focus   | `(options?: FocusOptions) => void` | -       |
| blur  | Lose focus  | `() => void`                       | -       |
