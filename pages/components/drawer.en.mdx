### Basic usage

<Code 
  description='Click the button, and the drawer will slide out from the right side. You can close the drawer by clicking on the overlay or the close button.' 
  component={<Basic />}
>
```tsx filename="Drawer.tsx" copy
import React from "react";
import { Drawer, Button } from "@ltopx/lx-ui";

export default function () {
  const [open, setOpen] = React.useState(false);
  return (
    <div>
      <Button type="primary" onClick={() => setOpen(true)}>
        Drawer
      </Button>
      <Drawer title="This is a title." open={open} onClose={() => setOpen(false)}>
        <div>Drawer Content</div>
      </Drawer>
    </div>
  );
}
```
</Code>

### Custom Width

<Code component={<Width />}>
<Code.Description>
Pass in `width` to customize the width of the drawer. The default width is `378px`.
</Code.Description>
```tsx filename="Drawer.tsx" copy {14}
import React from "react";
import { Drawer, Button } from "@ltopx/lx-ui";

export default function () {
  const [open, setOpen] = React.useState(false);
  
  return (
    <div className="flex">
      <Button type="primary" onClick={() => setOpen(true)}>
        Drawer
      </Button>
      <Drawer
        title="This is a title."
        width={500}
        open={open}
        onClose={() => setOpen(false)}
      >
        <div>Drawer Content</div>
      </Drawer>
    </div>
  );
}
```
</Code>

### Location

<Code 
  description='Customize the position of the drawer popup. Currently, it supports popping out from both the left and right sides.' 
  component={<Placement />}
>
```tsx filename="Drawer.tsx" copy {20,28}
import React from "react";
import { Drawer, Button } from "@ltopx/lx-ui";

export default function () {
  const [openLeft, setOpenLeft] = React.useState(false);
  const [openRight, setOpenRight] = React.useState(false);
  
  return (
    <>
      <div className="flex gap-2">
        <Button type="primary" onClick={() => setOpenLeft(true)}>
          Left Drawer
        </Button>
        <Button type="primary" onClick={() => setOpenRight(true)}>
          Right Drawer
        </Button>
      </div>
      <Drawer
        title="This is a title."
        placement="left"
        open={openLeft}
        onClose={() => setOpenLeft(false)}
      >
        <div>Drawer Content</div>
      </Drawer>
      <Drawer
        title="This is a title."
        placement="right"
        open={openRight}
        onClose={() => setOpenRight(false)}
      >
        <div>Drawer Content</div>
      </Drawer>
    </>
  );
}
```
</Code>

## API

### Drawer Props

| Name             | Description                                              | Type                 | Default |
| ---------------- | -------------------------------------------------------- | -------------------- | ------- |
| className        | Additional css class                                     | `string`             | -       |
| overlayClassName | Mask layer class                                         | `string`             | -       |
| title            | Custom title                                             | `React.ReactNode`    | Title   |
| placement        | Drawer Pull-out Direction                                | `left` \| `right`    | `right` |
| maskClosable     | Whether can be closed by click mask                      | `boolean`            | true    |
| open             | Whether to display                                       | `boolean`            | -       |
| width            | The width of the drawer                                  | `number` \| `string` | 378     |
| onClose          | The callback function triggered when closing the drawer. | `() => void`         | -       |
