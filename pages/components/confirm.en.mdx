### Basic usage

<Code component={<Basic />}>
<Code.Description>
Provide five basic types of confirmation dialogs that are the same as the [Button](/components/button) component. Trigger the `onOk` callback when the "Confirm" button is clicked.
</Code.Description>
```tsx filename="Confirm.tsx" copy
import { Confirm, Button } from "@ltopx/lx-ui";

export default function () {
  const content =
    "Lorem, ipsum dolor sit amet consectetur adipisicing elit. Eum quo ab quis odio maxime, distinctio ea nesciunt provident cumque tempora libero officia fuga accusamus consequuntur ipsum recusandae facere voluptatem sapiente.";
  const onOk = () => {
    console.log("ok");
  };
  
  const asyncOk = () => {
    return new Promise((resolve) => {
      setTimeout(() => {
        resolve("asyncOk");
      }, 2000);
    });
  };
  
  return (
    <div className="flex flex-wrap gap-x-3 gap-y-2">
      <Confirm
        title="This is a title."
        content={content}
        type="primary"
        onOk={asyncOk}
      >
        <Button type="primary">Confirm</Button>
      </Confirm>
      <Confirm
        title="This is a title."
        content={content}
        disabled
        type="primary"
        onOk={onOk}
      >
        <Button type="primary" disabled>
          Confirm
        </Button>
      </Confirm>
      <Confirm
        title="This is a title."
        content={content}
        type="success"
        onOk={onOk}
      >
        <Button type="success">Confirm</Button>
      </Confirm>
      <Confirm
        title="This is a title."
        content={content}
        type="warning"
        onOk={onOk}
      >
        <Button type="warning">Confirm</Button>
      </Confirm>
      <Confirm
        title="This is a title."
        content={content}
        type="danger"
        onOk={onOk}
      >
        <Button type="danger">Confirm</Button>
      </Confirm>
    </div>
  );
}
```
</Code>

## API

### Confirm Props

| Name             | Description                                | Type                                                         | Default   |
| ---------------- | ------------------------------------------ | ------------------------------------------------------------ | --------- |
| className        | Additional css class                       | `string`                                                     | -         |
| overlayClassName | Mask layer class                           | `string`                                                     | -         |
| maskClosable     | Whether can be closed by click mask        | `boolean`                                                    | true      |
| disabled         | Whether is disabled                        | `boolean`                                                    | -         |
| type             | Confirm type                               | `default` \| `primary` \| `success` \| `warning` \| `danger` | `primary` |
| title            | Custom title                               | `React.ReactNode`                                            | Title     |
| icon             | Custom title icon                          | `React.ReactNode`                                            | -         |
| content          | Custom content                             | `React.ReactNode`                                            | Content   |
| okText           | Custom the text of the confirmation button | `React.ReactNode`                                            | Ok        |
| cancelText       | Custom the text of the cancel button       | `React.ReactNode`                                            | Cancel    |
| onOk             | Click the "Confirm" button callback        | `(e: Event) => void`                                         | -         |
