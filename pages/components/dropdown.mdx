### 支持可选中

<Code 
  description='可以支持选中，在某些场景下可以替代 Select 使用。' 
  component={<Selectable />}
>
```tsx filename="Dropdown.tsx" copy
import React from "react";
import { Drawer, Button, type DropdownOption } from "@ltopx/lx-ui";

export default function () {
  const options: DropdownOption[] = [
    { label: "土豆,土豆", value: "111" },
    { label: "西红柿,西红柿", value: "222;" },
    { label: "马铃薯,马铃薯", value: "333" },
    { label: "番茄,番茄", value: "444" },
  ];
  
  const [value, setValue] = React.useState("111");
  
  return (
    <div className="flex flex-wrap gap-2">
      <Dropdown options={options} selectable value={value} onSelect={setValue}>
        <Button type="primary">Selectable</Button>
      </Dropdown>
    </div>
  );
}
```
</Code>

## API

### Dropdown Props

| 属性        | 描述               | 类型                         | 默认值   |
| ----------- | ------------------ | ---------------------------- | -------- |
| className   | 内容区域类名       | `string`                     | -        |
| disabled    | 是否禁用           | `boolean`                    | -        |
| side        | 弹出方向           | `top` \| `bottom`            | `bottom` |
| sideOffset  | 弹出方向的间距     | `number`                     | 6        |
| align       | 对齐方向           | `start` \| `center` \| `end` | `center` |
| alignOffset | 对齐方向的间距     | `number`                     | -        |
| label       | 标题               | `string`                     | -        |
| options     | 选项列表           | `DropdownOption[]`           | -        |
| selectable  | 是否可支持选中     | `boolean`                    | -        |
| value       | 当前选中的值       | `string` \| `number`         | -        |
| onSelect    | 选中某一选项的回调 | `(value: any) => void`       | -        |

### DropdownOption Props

| 属性  | 描述           | 类型              | 默认值 |
| ----- | -------------- | ----------------- | ------ |
| label | 选项展示内容   | `React.ReactNode` | -      |
| value | 选项选中时的值 | `string`          | -      |
| type  | 是否禁用       | `seperate`        | -      |
