### Controlled mode

<Code 
  description='Enable controlled mode through activeTab.' 
  component={<Active />}
>
```tsx filename="Tabs.tsx" copy
import React from "react";
import { Tabs, type TabsOption } from "@ltopx/lx-ui";

export default function () {
  const [active, setActive] = React.useState("1");
  
  const options: TabsOption[] = [
    {
      label: "Tab1",
      value: "1",
      children: <div>xxxxx</div>,
    },
    {
      label: "Tab2",
      value: "2",
      children: <div>yyyyy</div>,
    },
  ];
  
  return (
    <div className="flex flex-col gap-2">
      <div>{active}</div>
      <Tabs options={options} activeTab={active} onChange={setActive} />
    </div>
  );
}
```
</Code>

## API

### Tabs Props

| Name             | Description                                 | Type                      | Default |
| ---------------- | ------------------------------------------- | ------------------------- | ------- |
| className        | Additional css class                        | `string`                  | -       |
| size             | Tabs size                                   | `sm` \| `base` \| `lg`    | `base`  |
| options          | Tabs options                                | `TabsOption[]`            | -       |
| defaultActiveTab | Default active item                         | `string`                  | -       |
| itemsFull        | Whether the tab options occupy a full line. | `boolean`                 | -       |
| activeTab        | Selected label options in controlled mode.  | `number`                  | -       |
| onChange         | Callback when switching tabs                | `(value: string) => void` | -       |

### TabsOption Props

| Name     | Description                                 | Type              | Default |
| -------- | ------------------------------------------- | ----------------- | ------- |
| label    | Tab Display Text                            | `React.ReactNode` | -       |
| value    | Select the corresponding value for the tab. | `string`          | -       |
| children | Tab Display Content                         | `React.ReactNode` | -       |
