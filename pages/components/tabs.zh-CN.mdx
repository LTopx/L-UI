### 受控模式

<Code 
  description='通过 activeTab 开启受控模式。' 
  component={<Active />}
>
```tsx filename="Tabs.tsx" copy
import React from "react";
import { Tabs, type TabsOption } from "@ltopx/lx-ui";

export default function () {
  const [active, setActive] = React.useState("1");
  
  const options: TabsOption[] = [
    {
      label: "Tab1",
      value: "1",
      children: <div>xxxxx</div>,
    },
    {
      label: "Tab2",
      value: "2",
      children: <div>yyyyy</div>,
    },
  ];
  
  return (
    <div className="flex flex-col gap-2">
      <div>{active}</div>
      <Tabs options={options} activeTab={active} onChange={setActive} />
    </div>
  );
}
```
</Code>

## API

### Tabs Props

| 属性             | 描述                     | 类型                      | 默认值 |
| ---------------- | ------------------------ | ------------------------- | ------ |
| className        | 节点类名                 | `string`                  | -      |
| size             | 尺寸大小                 | `sm` \| `base` \| `lg`    | `base` |
| options          | Tabs 选项内容            | `TabsOption[]`            | -      |
| defaultActiveTab | 默认选中的标签选项       | `string`                  | -      |
| itemsFull        | 标签页选项是否占满一行   | `boolean`                 | -      |
| activeTab        | 受控模式下选中的标签选项 | `number`                  | -      |
| onChange         | 切换标签页时的回调       | `(value: string) => void` | -      |

### TabsOption Props

| 属性     | 描述             | 类型              | 默认值 |
| -------- | ---------------- | ----------------- | ------ |
| label    | 标签页显示文本   | `React.ReactNode` | -      |
| value    | 标签页选中对应值 | `string`          | -      |
| children | 标签页显示内容   | `React.ReactNode` | -      |
